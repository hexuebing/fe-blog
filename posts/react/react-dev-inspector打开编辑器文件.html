                                        # react-dev-inspector打开编辑器文件

                                        # 安装包

                                        npm i -D react-dev-inspector

                                        # 在根文件中注入

                                        import { createRoot } from 'react-dom/client'
import { Inspector } from 'react-dev-inspector'
import { App } from './App'
import './index.css'
 
createRoot(document.getElementById('root') as HTMLDivElement)
  .render(
    <React.StrictMode>
      <Inspector />
      <App />
    </React.StrictMode>,
  )
