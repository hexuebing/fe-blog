var lengthOfLongestSubstring = function(s) {
    let left = 0, right = 0;
    const window = {}
    let result = 0
    while(right < s.length){
        const c = s[right]
        window[c] = window[c] ? window[c] + 1 : 1
        right++
        while(window[c]>1){
            const f = s[left]
            left++
            window[f]--
        }
        result = Math.max(result, right-left)
    }
    return result
};
