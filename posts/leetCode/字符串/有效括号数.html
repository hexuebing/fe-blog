                        # 有效括号

                        给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串，判断字符串是否有效。

                        有效字符串需满足：

                        左括号必须用相同类型的右括号闭合。 左括号必须以正确的顺序闭合。 注意空字符串可被认为是有效字符串。

                        示例 1:
输入: "()"
输出: true
示例 2:
输入: "()[]{}"
输出: true
示例 3:
输入: "(]"
输出: false
示例 4:
输入: "([)]"
输出: false
示例 5:
输入: "{[]}"
输出: true

                        # 我的解法：

                        /**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function(s) {
    if(s.length == 0) return false
    let sArr = s.split('')
    let sSetArr = [...new Set(sArr)]
    if(sArr.length == sSetArr.length){
        if(sSetArr.length % 2 != 1){
            let flag = false
            if(sSetArr.includes('(')){
                flag = sSetArr.includes(')') ? true : false
            }
            if(sSetArr.includes('[')){
                flag = sSetArr.includes(']') ? true : false
            }
            if(sSetArr.includes('{')){
                flag = sSetArr.includes('}') ? true : false
            }
            return flag
        }
    }
    return false
};

        switch(letter) {
            case "(": {
                arr.push(letter)
                break;
            }
            case "[": {
                arr.push(letter)
                break;
            }
            case "{": {
                arr.push(letter)
                break;
            }

            }

    }
