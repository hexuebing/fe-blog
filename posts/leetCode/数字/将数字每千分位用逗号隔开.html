  let int = ''

  }
  const end = decimals ? '.'+decimals : ''
  if(int.length < 3){
    return n
  }else{
    let left = int.length % 3
    if(left > 0){
      return int.slice(0, left) + ',' + int.slice(left).match(/\d{3}/g).join(',') + end
    }else{
      return int.slice(left).match(/\d{3}/g).join(',') + end
    }
  }
