# 二叉树遍历

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
```

## 二叉树先序遍历

### 递归方法

```js
var preorderTraversal = function (root){
    let result = []
    function travel(curNode){
        curNode && result.push(curNode.val)
        if(curNode && curNode.left){
            travel(curNode.left)
        }
        if(curNode && curNode.right){
            travel(curNode.right)
        }
    }
    travel(root)
    return result
}
```

### 栈方法

```js
var preorderTraversal = function(root) {
    let result = []
    let task = []
    if(root) task.push(root)
    while(task.length>0){
        let node = task.pop()
        result.push(node.val)
        node.right && task.push(node.right)
        node.left && task.push(node.left)
    }
    return result
};
```

## 二叉树中序遍历

```js
var inorderTraversal = function(root) {
    let result = []
    function travel(curNode){
        if(curNode && curNode.left) {
            travel(curNode.left)
        }
        curNode && result.push(curNode.val)
        if(curNode && curNode.right){
            travel(curNode.right)
        } 
    }
    travel(root)
    return result
}
```

## 二叉树后序遍历

### 递归实现

```js
var postorderTraversal = function (root){
    let result = []
    function travel(cur){
        if(cur != null){
            if(cur.left){
                travel(cur.left)
            }
            if(cur.right){
                travel(cur.right)
            }
            result.push(cur.val)
        }
    }
    travel(root)
    return result
}
```

### 极简代码

```js
var postorderTraversal = function(root) {
    return root ? [...postorderTraversal(root.left), ...postorderTraversal(root.right), root.val] : []
};
```
