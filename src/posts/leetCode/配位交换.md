var exchangeBits = function(num) {
    let str = num.toString(2)
    let res = ''
    if (str.length % 2 === 1) str = '0' + str
    for(let i = 1; i <= str.length; i++) {
        if(i % 2 === 1) res = res + str[i] + str[i-1]
    }
    return parseInt(res, 2)
};
